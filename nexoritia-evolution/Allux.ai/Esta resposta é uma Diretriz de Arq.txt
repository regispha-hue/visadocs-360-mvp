1. MODELO DE DADOS: O SISTEMA DE WORKSPACES
O sistema deve abandonar o arquivo único e adotar uma estrutura de Diretório de Contextos. Cada obra (Antinimia, Anytia, LDM) deve ser uma instância isolada.

Estrutura de Armazenamento: Implementar um WorkspaceManager que gerencie bancos de dados SQLite independentes para cada projeto (/memory/projects/{project_name}.db).

Ingestão Dinâmica: O sistema deve aceitar comandos para montar volumes a partir de arquivos ZIP.

POST /workspace/ingest: Descompacta o ZIP, indexa os fragmentos em um banco de vetores (RAG) e extrai o Kernel inicial.

2. A CAMADA EGO-OS (MIDDLEWARE DE IDENTIDADE)
O Ego-OS deve ser implementado como a Camada Zero. Ele não contém enredo, mas sim a Ontologia Estilística do autor.

Implementação: O Ego-OS atua como um IdentityMiddleware. Antes de qualquer requisição à API de IA, ele injeta um Vetor de Estilo (Visceralidade, Veniloquismo, Assinatura) no System Prompt.

Prioridade: Ego-OS (Estilo) > LDMux-Kernel (Lei da Obra) > User Command (Ação).

3. MOTOR DE VENTILOQUISMO (RAG & CONTEXTO)
Para preencher lacunas nos Montes/Capítulos, o sistema deve ser capaz de:

Análise de Vizinhança: Recuperar prosa já escrita ao redor da lacuna (Context-Awareness).

Injeção de Oralidade: Utilizar documentos de transcrição de áudio (se presentes no ZIP) para ajustar o tom da IA para a voz real do autor.

4. ORQUESTRADOR MULTI-MODELO & CLOUD
O servidor deve ser refatorado para ser Stateless (via FastAPI), permitindo deploy em nuvem (Docker/AWS/DigitalOcean).

Portabilidade: O "Executável" deve ser um cliente CLI leve que se comunica via HTTPS com o servidor central.

Driver Unificado: Criar uma interface BaseLLM que padronize a comunicação com GPT-4o, Claude 3.5 e Gemini 1.5 Pro, garantindo que o Halt Ontológico (Silêncio) seja respeitado por todos.

# orchestrator/workspace_manager.py
from __future__ import annotations
import os
import sqlite3
from dataclasses import dataclass
from pathlib import Path
from typing import Optional, Iterable, Dict, Any

DEFAULT_BASE = Path("storage/projects")

SCHEMA_SQL = """
CREATE TABLE IF NOT EXISTS meta (
  key TEXT PRIMARY KEY,
  value TEXT NOT NULL
);

CREATE TABLE IF NOT EXISTS fragments (
  id INTEGER PRIMARY KEY AUTOINCREMENT,
  path TEXT NOT NULL,
  content TEXT NOT NULL,
  created_at TEXT DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE IF NOT EXISTS outputs (
  id INTEGER PRIMARY KEY AUTOINCREMENT,
  command TEXT NOT NULL,
  output TEXT NOT NULL,
  status TEXT NOT NULL,         -- draft|frozen|archived
  created_at TEXT DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE IF NOT EXISTS kernel (
  id INTEGER PRIMARY KEY AUTOINCREMENT,
  name TEXT NOT NULL,           -- laws|other
  yaml TEXT NOT NULL,
  created_at TEXT DEFAULT CURRENT_TIMESTAMP
);
"""

@dataclass
class Workspace:
  name: str
  db_path: Path

class WorkspaceManager:
  def __init__(self, base_dir: Path = DEFAULT_BASE):
    self.base_dir = base_dir
    self.base_dir.mkdir(parents=True, exist_ok=True)

  def _db_path(self, name: str) -> Path:
    safe = name.strip().replace(" ", "_")
    return self.base_dir / f"{safe}.db"

  def create(self, name: str) -> Workspace:
    db = self._db_path(name)
    conn = sqlite3.connect(db)
    try:
      conn.executescript(SCHEMA_SQL)
      conn.execute("INSERT OR REPLACE INTO meta(key,value) VALUES(?,?)", ("workspace_name", name))
      conn.commit()
    finally:
      conn.close()
    return Workspace(name=name, db_path=db)

  def exists(self, name: str) -> bool:
    return self._db_path(name).exists()

  def get(self, name: str) -> Workspace:
    db = self._db_path(name)
    if not db.exists():
      raise FileNotFoundError(f"Workspace '{name}' não existe.")
    return Workspace(name=name, db_path=db)

  def connect(self, ws: Workspace) -> sqlite3.Connection:
    conn = sqlite3.connect(ws.db_path)
    conn.row_factory = sqlite3.Row
    return conn

  def set_kernel_yaml(self, ws: Workspace, kernel_name: str, yaml_text: str) -> None:
    conn = self.connect(ws)
    try:
      conn.execute("INSERT INTO kernel(name,yaml) VALUES(?,?)", (kernel_name, yaml_text))
      conn.commit()
    finally:
      conn.close()

  def latest_kernel_yaml(self, ws: Workspace, kernel_name: str) -> Optional[str]:
    conn = self.connect(ws)
    try:
      row = conn.execute(
        "SELECT yaml FROM kernel WHERE name=? ORDER BY id DESC LIMIT 1",
        (kernel_name,)
      ).fetchone()
      return None if row is None else str(row["yaml"])
    finally:
      conn.close()

  def add_fragments(self, ws: Workspace, items: Iterable[Dict[str, str]]) -> int:
    conn = self.connect(ws)
    try:
      cur = conn.cursor()
      cur.executemany(
        "INSERT INTO fragments(path, content) VALUES(?,?)",
        [(it["path"], it["content"]) for it in items]
      )
      conn.commit()
      return cur.rowcount
    finally:
      conn.close()

  def save_output(self, ws: Workspace, command: str, output: str, status: str = "draft") -> int:
    conn = self.connect(ws)
    try:
      cur = conn.cursor()
      cur.execute(
        "INSERT INTO outputs(command, output, status) VALUES(?,?,?)",
        (command, output, status)
      )
      conn.commit()
      return int(cur.lastrowid)
    finally:
      conn.close()

# orchestrator/identity_middleware.py
from __future__ import annotations
import yaml
from dataclasses import dataclass
from typing import Dict, Any, Optional

@dataclass(frozen=True)
class EgoIdentity:
  author: str
  style_vector: Dict[str, float]
  tone_rules: list[str]
  forbidden_patterns: list[str]
  halt_token: str

def load_ego_yaml(path: str = "kernel/ego_os.yaml") -> EgoIdentity:
  with open(path, "r", encoding="utf-8") as f:
    data = yaml.safe_load(f)
  ident = data.get("identity", {})
  return EgoIdentity(
    author=str(ident.get("author", "Unknown")),
    style_vector=dict(ident.get("style_vector", {})),
    tone_rules=list(ident.get("tone_rules", [])),
    forbidden_patterns=list(ident.get("forbidden_patterns", [])),
    halt_token=str(ident.get("halt_token", "Silêncio")),
  )

def build_ego_system_block(ego: EgoIdentity) -> str:
  vec = ", ".join([f"{k}={v:.2f}" for k, v in ego.style_vector.items()])
  rules = "\n".join([f"- {r}" for r in ego.tone_rules])
  forb = "\n".join([f"- {p}" for p in ego.forbidden_patterns])
  return (
    "EGO-OS (Camada Zero) — Identidade Imutável\n"
    f"Autor: {ego.author}\n"
    f"Vetor de Estilo: {vec}\n"
    "Regras de Tom:\n"
    f"{rules}\n"
    "Proibições:\n"
    f"{forb}\n"
    f"HALT_TOKEN: {ego.halt_token}\n"
  )

def build_system_prompt(ego: EgoIdentity, laws_yaml: str, extra_context: str = "") -> str:

    "- Não inventar leis fora do Kernel.\n"

  req = LLMRequest(system=system, user=user_command)

6) orchestrator/dsm_guard.py (HALT + proibições)
# orchestrator/dsm_guard.py
from __future__ import annotations
import re
import yaml

def _extract_forbidden_from_kernel(laws_yaml: str) -> list[str]:
  try:
    data = yaml.safe_load(laws_yaml) or {}
    forb = data.get("forbidden", [])
    return [str(x) for x in forb]
  except Exception:
    return []

def enforce_halt_and_forbidden(text: str, halt_token: str, laws_yaml: str) -> str:
  # HALT detect: se o output contém a Palavra Final, corta e encerra
  if halt_token and halt_token.lower() in text.lower():
    # Mantém até o token (primeira ocorrência) e encerra
    idx = text.lower().find(halt_token.lower())
    return text[: idx + len(halt_token)].strip()

  forbidden = _extract_forbidden_from_kernel(laws_yaml)
  for pat in forbidden:
    if pat and re.search(re.escape(pat), text, flags=re.IGNORECASE):
      raise RuntimeError(f"VIOLAÇÃO_Kernel: padrão proibido detectado: {pat}")

  return text

# memory/vector_engine.py
from __future__ import annotations
from typing import List, Dict, Any, Optional
import sqlite3

class VectorEngine:
  """
  v0: busca lexical + vizinhança.
  v1: plugar embeddings (FAISS/Chroma) sem mudar a API.
  """
  def __init__(self, conn: sqlite3.Connection):
    self.conn = conn

  def search(self, query: str, limit: int = 8) -> List[Dict[str, Any]]:
    q = f"%{query}%"
    rows = self.conn.execute(
      "SELECT id, path, content FROM fragments WHERE content LIKE ? LIMIT ?",
      (q, limit)
    ).fetchall()
    return [{"id": r["id"], "path": r["path"], "content": r["content"]} for r in rows]

  def neighborhood(self, fragment_id: int, window: int = 2) -> List[Dict[str, Any]]:
    # pega fragmentos próximos por id (simplificação)
    lo = max(1, fragment_id - window)
    hi = fragment_id + window
    rows = self.conn.execute(
      "SELECT id, path, content FROM fragments WHERE id BETWEEN ? AND ? ORDER BY id ASC",
      (lo, hi)
    ).fetchall()
    return [{"id": r["id"], "path": r["path"], "content": r["content"]} for r in rows]

from __future__ import annotations
import io
import zipfile
from pathlib import Path
from fastapi import FastAPI, UploadFile, File, HTTPException
from pydantic import BaseModel

from orchestrator.workspace_manager import WorkspaceManager
from orchestrator.identity_middleware import load_ego_yaml
from memory.vector_engine import VectorEngine

app = FastAPI(title="LDMux-OS v2.0")

wm = WorkspaceManager()
EGO = load_ego_yaml()
WORKSPACE_STATE = {"active": None}  # stateless real: isso vai para token/cookie/headers em cloud

class WorkspaceCreate(BaseModel):
  name: str

class WorkspaceSwitch(BaseModel):
  name: str

class CommandRun(BaseModel):
  workspace: str
  command: str

@app.post("/workspace/create")
def create_ws(payload: WorkspaceCreate):
  ws = wm.create(payload.name)
  return {"ok": True, "workspace": ws.name}

@app.post("/workspace/switch")
def switch_ws(payload: WorkspaceSwitch):
  if not wm.exists(payload.name):
    raise HTTPException(404, f"Workspace '{payload.name}' não existe.")
  WORKSPACE_STATE["active"] = payload.name
  return {"ok": True, "active": payload.name}

@app.post("/workspace/ingest")
async def ingest_zip(workspace: str, zip_file: UploadFile = File(...)):
  if not wm.exists(workspace):
    wm.create(workspace)
  ws = wm.get(workspace)

  raw = await zip_file.read()
  extract_dir = Path("storage/workspaces") / workspace
  extract_dir.mkdir(parents=True, exist_ok=True)

  fragments = []
  laws_yaml_text = None

  with zipfile.ZipFile(io.BytesIO(raw)) as z:
    for name in z.namelist():
      if name.endswith("/") or name.startswith("__MACOSX"):
        continue
      content = z.read(name)
      out_path = extract_dir / name
      out_path.parent.mkdir(parents=True, exist_ok=True)
      out_path.write_bytes(content)

      # heurística: capturar laws.yaml se existir
      if name.lower().endswith("laws.yaml"):
        laws_yaml_text = content.decode("utf-8", errors="ignore")
        continue

      # indexar textos simples (md/txt)
      if name.lower().endswith((".md", ".txt")):
        fragments.append({
          "path": name,
          "content": content.decode("utf-8", errors="ignore")
        })

  if fragments:
    wm.add_fragments(ws, fragments)

  if laws_yaml_text:
    wm.set_kernel_yaml(ws, "laws", laws_yaml_text)

  return {
    "ok": True,
    "workspace": workspace,
    "fragments_indexed": len(fragments),
    "kernel_detected": bool(laws_yaml_text)
  }

@app.get("/workspace/search")
def search(workspace: str, q: str, limit: int = 8):
  ws = wm.get(workspace)
  conn = wm.connect(ws)
  try:
    ve = VectorEngine(conn)
    hits = ve.search(q, limit=limit)
    return {"ok": True, "hits": hits}
  finally:
    conn.close()
